		\begin{frame}[fragile]{Expressões de Atribuição}
			\begin{block}{}
				\begin{itemize}
					\item Fundamental para todas as linguagens imperativas.
				\end{itemize}
			\end{block} 
		
\begin{lstlisting}[style=C++]
	a = b;		//C, Fortran			
	c := d;		//ALGOL
\end{lstlisting} \pause
			
			\begin{block}{Semântica de Cópia}
				\begin{itemize}
					\item expressão avaliada para um valor;
					\item valor copiado para o destino;
				\end{itemize}
			\end{block}
		\end{frame}
	
		\begin{frame}[fragile]{Expressões de Atribuição}
\begin{lstlisting}[style=C++]
	#include <math.h>
	
	bool a;
	double b;
		
	a = (5 > 3) || (2*3 == 4);		// valor de 'a': true
	b = pow(2.0, 3.0); 					// valor de 'b': 8.0		
\end{lstlisting}
			
			\begin{block}{Expressões}

			\begin{block}{Sequência}
				\begin{itemize}
					\item Fluxo natural de execução.
				\end{itemize}
\begin{lstlisting}[style=C++]
	a = 5 > b;
	b = c;
	c++;		
\end{lstlisting} \pause
				\begin{itemize}
					\item Comandos de desvios: {\bf return}, {\bf break}, {\bf continue}, e {\bf go to}.
				\end{itemize} \pause
\begin{lstlisting}[style=C++]

	rotulo1:
		int a = 10;
		a = a - 2;
		goto rotulo1;
\end{lstlisting}

		\begin{frame}[fragile]{Estruturas de controle}
			\begin{block}{Condicional}
				\begin{itemize}
					\item Seleciona caminhos alternativos em execução
				\end{itemize}
\begin{lstlisting}[style=C++]
	if(a > b){
		a++;
	}
	
	switch(a){
		case 5:
			a--;
			break;
		case 6:
			a = 3;
			break;
		default: a = 1;
	}
\end{lstlisting}
			\end{block}
		\end{frame}
		
		\begin{frame}[fragile]{Estruturas de controle}
			\begin{block}{Laço}
				\begin{itemize}
					\item Duas variações: teste no início ou no fim; \pause
					\item Em C, é chamado de {\bf while}; \pause
					\item O comando {\bf for} é uma variação bastante utilizada; \pause
					\item Comandos de desvios podem ser utilizados.
				\end{itemize}
\begin{lstlisting}[style=C++]
	while(a > b){
		a++;
	}
	
	do{
		a++;
	}while(a > b)
\end{lstlisting}
			\end{block}
		\end{frame}
	
		\begin{frame}[fragile]{Estruturas de controle}
			\begin{block}{Laço}
				\begin{itemize}
					\item O comando {\bf for} é uma variação bastante utilizada;
				\end{itemize}
\begin{lstlisting}[style=C++]
	int i, a;
	
	a=0;
	for(i=0; i<5; i++){
		if(i % 4 == 0)
			continue;
		a = a + i;
	}	
	while(a > 0){
		if(a == 2)
			break;
		a--;
	}	
\end{lstlisting}
			\end{block}
		\end{frame}

			\begin{block}{Necessidades}
				\begin{itemize}
					\item Meio de lidar com erros inesperados: \pause
						\begin{itemize}
							\item Tempo de Execução; \pause
							\item Entrada e saída; \pause
							\item Acesso à memória, etc.
						\end{itemize} \pause
					\item Permite manipular erros ao invés de abortar a execução; \pause
					\item Níveis de exceções: \pause
					\begin{itemize}
						\item Hardware (e.g. divisão por zero); \pause
						\item Linguagem de Programação (e.g. índice fora do intervalo); \pause
						\item Outros níveis (e.g. desempilhar pilha vazia).
					\end{itemize}
				\end{itemize}

		\begin{frame}{Potencialidades}
			\begin{block}{Pontos fortes}
				\begin{itemize}
					\item Exigência de alto desempenho; \pause
					\item Programação mais próxima ao nível de máquina; \pause
					\begin{itemize}
						\item e.g. C é tratada como linguagem de máquina universal. \pause
					\end{itemize}
					\item Restrições de memória ou potência. \pause
					\begin{itemize}
						\item e.g. programação em dispositivos móveis.
					\end{itemize}
				\end{itemize}
			\end{block}
		\end{frame}
		
		\begin{frame}{Fragilidades}
			\begin{block}{Limitações da Decomposição Funcional}
				\begin{itemize}
					\item Muitas aplicações não são bem servidas neste paradigma; \pause
					\item Programação de GUIs e dispositivos embarcados; \pause
					\item Outra abordagem: troca de mensagens entre objetos; \pause
					\item Programação de GUIs:
					\begin{itemize}
						\item  coleção de diferentes tipos de
						objetos ? botões, áreas de texto, imagens, videoclipes e menus desdobráveis
						?, cada um comunicando-se com o programa e com o usuário, mandando e
						recebendo mensagens. \pause
						\item ``inventam a roda''. 
					\end{itemize} \pause
					\item Diálogo entre os {\it stakeholders}.
				\end{itemize}
			\end{block}
